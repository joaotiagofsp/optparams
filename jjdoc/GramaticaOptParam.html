NON-TERMINALS

Input ::= PPrograma <EOF>
PValorInteiro ::= <INTEGER_LITERAL>
PValorBooleano ::= <FALSE>
| <TRUE>
PValorString ::= <STRING_LITERAL>
PValor ::= ( PValorInteiro | PValorBooleano | PValorString )
PId ::= <IDENTIFIER>
PExpMenos ::= <MINUS> PExpPrimaria
PExpNot ::= <NOT> PExpPrimaria
PExpLength ::= <LENGTH> PExpPrimaria
PExpPrimaria ::= ( PValor | PId | <LPAREN> PExpressao <RPAREN> )
PDeclVar ::= ( <VAR> PId <ASSIGN> PExpressao ) 
///////////////////////////////////////////////////////////////////////////////
//////////////////////    REGRA DE PRODUCAO ATUALIZADA   //////////////////////
///////////////////////////////////////////////////////////////////////////////
PDeclFuncao ::= ( <FUNC> PId PListaParametro <ASSIGN> PExpressao )
PDecComposta ::= ( PDeclVar | PDeclFuncao ) <COMMA> PDeclFuncional
PDeclFuncional ::= ( PDecComposta | PDecComposta | PDeclVar | PDeclFuncao | <LPAREN> PDeclFuncional <RPAREN> )
PExpDeclaracao ::= <LET> PDeclFuncional <IN> PExpressao
PExpCondicional ::= <IF> PExpressao <THEN> PExpressao <ELSE> PExpressao
PAplicacao ::= ( PId <LPAREN> <RPAREN> | PId <LPAREN> PListaExpr <RPAREN> ) 
/**
* Parametro ::= ParametroObrigatorio | ParametroOpcional
**/
PParametro ::= ( PParametroOpcional | PParametroObrigatorio ) 
/**
* ParametroObrigatorio ::= Id
**/
PParametroObrigatorio ::= PId 
/**
* ParametroOpcional ::= Id "?" "(" Expressao ")"
**/
PParametroOpcional ::= PId <HOOK> <LPAREN> PExpressao <RPAREN> 
/**
* Armazena a lista de parâmetros, que podem ser parâmetros obrigatórios ou opcionais. 
* PListaParametro ::= Parametro  |  Parametro  ListParametroOpcional
*/
PListaParametro ::= ( PParametro )+
PListaExpr ::= ( PExpressao ( <COMMA> PExpressao )* )
PExpUnaria ::= ( PExpMenos | PExpNot | PExpLength | PExpDeclaracao | PExpCondicional | PAplicacao | PExpPrimaria )
PExpBinaria ::= ( PExpBinaria2 ( <EQ> PExpBinaria2 )* )
PExpBinaria2 ::= ( PExpBinaria3 ( <PLUS> PExpBinaria3 | <MINUS> PExpBinaria3 | <OR> PExpBinaria3 | <CONCAT> PExpBinaria3 )* )
PExpBinaria3 ::= ( PExpUnaria ( <AND> PExpUnaria )* )
PExpressao ::= PExpBinaria
PPrograma ::= PExpressao
